SELECT CTYNAME, 
(CASE	WHEN POPESTIMATE2011 > POPESTIMATE2010
	THEN POPESTIMATE2011 - POPESTIMATE2010
	ELSE POPESTIMATE2010 - POPESTIMATE2011
	END
)AS CHANGE,
(CASE	WHEN POPESTIMATE2011 > POPESTIMATE2010
	THEN 'INCREASE'
	WHEN POPESTIMATE2011 < POPESTIMATE2010
	THEN 'DECREASE'
	WHEN POPESTIMATE2011 = POPESTIMATE2010
	THEN 'SAME'
	END
)AS INDICATOR
FROM POP_EST_STATE_COUNTY
WHERE STNAME = 'Wisconsin' AND SUMLEV = 50;
